rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /guestbook/{messageId} {
      allow read: if true;
      allow create: if request.auth != null
                    && request.auth.token.firebase.sign_in_provider == "anonymous"
                    && request.resource.data.keys().hasOnly([
                         "nickname", "message", "uid", "createdAt"
                       ])
                    && request.resource.data.nickname is string
                    && request.resource.data.nickname.size() >= 1
                    && request.resource.data.nickname.size() <= 20
                    && request.resource.data.message is string
                    && request.resource.data.message.size() >= 1
                    && request.resource.data.message.size() <= 300
                    && request.resource.data.uid == request.auth.uid;
      allow update, delete: if false;
    }
  }
}
